go 字符串拼接对比

November 18, 2022 · 2 min · 冲和
Table of Contents
  • 字符串拼接方式
  • 基准测试
  • 总结

字符串拼接方式#

  1. +=
  2. bytes.Buffer{}.WriteString()
  3. strings.Builder{}.WriteString()
  4. copy

基准测试#

代码：

var (
	str       = "s"
	count     = 1000
	expectStr = strings.Repeat(str, count)
)

func BenchmarkPlus(b *testing.B) {
	for i := 0; i < b.N; i++ {
		var s string
		for j := 0; j < count; j++ {
			s += str
		}
		if expectStr != s {
			b.Error("repeat count error")
		}
	}
}

func BenchmarkBuffer(b *testing.B) {
	for i := 0; i < b.N; i++ {
		var s bytes.Buffer
		for j := 0; j < count; j++ {
			s.WriteString(str)
		}
		if expectStr != s.String() {
			b.Error("repeat count error")
		}
	}
}

func BenchmarkBuilder(b *testing.B) {
	for i := 0; i < b.N; i++ {
		var s strings.Builder
		for j := 0; j < count; j++ {
			s.WriteString(str)
		}
		if expectStr != s.String() {
			b.Error("repeat count error")
		}
	}
}

func BenchmarkCopy(b *testing.B) {
	for i := 0; i < b.N; i++ {
		bs := make([]byte, len(str)*count)
		c := 0
		for j := 0; j < count; j++ {
			c += copy(bs[c:], str)
		}
		if expectStr != string(bs) {
			b.Error("repeat count error")
		}
	}
}

运行命令： go test mm/test -bench=. -benchmem -run=none

结果：

BenchmarkPlus-4            10101            119007 ns/op          530276 B/op        999 allocs/op
BenchmarkBuffer-4         199851              5407 ns/op            3008 B/op          6 allocs/op
BenchmarkBuilder-4        224800              4604 ns/op            3320 B/op          9 allocs/op
BenchmarkCopy-4           295381              3525 ns/op            1024 B/op          1 allocs/op

总结#

效率从高到低：

  • copy
  • strings.Builder
  • bytes.Buffer
  • +=
  • 字符串拼接
