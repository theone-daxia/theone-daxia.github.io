原子操作实现互斥锁

November 16, 2022 · 1 min · 冲和

下面的 MyMutex 即通过原子操作实现了 Locker 接口。

只是简易实现，实际的 Mutex 还会考虑到给 新来的goroutine、等待的goroutine更多机会， 还有饥饿处理（1ms后），防止队尾的goroutine总是得不到锁。

package test

import (
	"sync"
	"sync/atomic"
	"testing"
)

type MyMutex struct {
	key int32
}

func (m *MyMutex)Lock() {
	for !atomic.CompareAndSwapInt32(&m.key, 0, 1) {
	}
}

func (m *MyMutex)Unlock() {
	atomic.StoreInt32(&m.key, 0)
}

func TestMutex(t *testing.T)  {
	var count = 0
	var wg sync.WaitGroup
	var myMux = MyMutex{}

	wg.Add(10)
	for i := 0; i < 10; i++ {
		go func() {
			defer wg.Done()
			for i := 0; i < 10000; i++ {
				myMux.Lock()
				count++
				myMux.Unlock()
			}
		}()
	}
	wg.Wait()
	t.Log(count)
}
